\tree{Computation}{an executable I/O behavior}{
\tab\instance{Condition}{a [[:Category:Computation|computation]] that checks a condition on given data}
\tab\instance{Impure computation}{a computation that computes the result with side effects on data}
\tab\instance{Pure computation}{a computation that computes the result without side effects on data}
\tab\instance{Query}{a [[:Category:Computation|computation]] that performs a query on compound data}
\tab\concept{Transformation}{}
\tab\tab\instance{Endogenous transformation}{a transformation with the type of the input and output being the same}
\tab\tab\instance{Exogenous transformation}{a transformation with the type of the input and output being ''not'' the same}
\tab\tab\instance{Horizontal transformation}{a transformation that preserves the level of abstraction}
\tab\tab\instance{Vertical transformation}{a transformation that lowers or raises the level of abstraction}
}
